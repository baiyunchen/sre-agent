## 1. 事件总线

### 1.4 事件总线实现

```csharp
public interface IEventBus
{
    Task PublishAsync<T>(T @event) where T : IAgentEvent;
    IDisposable Subscribe<T>(Func<T, Task> handler) where T : IAgentEvent;
    IDisposable Subscribe(string eventType, Func<IAgentEvent, Task> handler);
}

public class InMemoryEventBus : IEventBus
{
    private readonly ConcurrentDictionary<Type, List<Delegate>> _handlers = new();
    private readonly ConcurrentDictionary<string, List<Func<IAgentEvent, Task>>> _typeHandlers = new();
    
    public async Task PublishAsync<T>(T @event) where T : IAgentEvent
    {
        // 类型化处理器
        if (_handlers.TryGetValue(typeof(T), out var handlers))
        {
            foreach (var handler in handlers.ToList())
            {
                await ((Func<T, Task>)handler)(@event);
            }
        }
        
        // 基于事件类型的处理器
        if (_typeHandlers.TryGetValue(@event.EventType, out var typeHandlers))
        {
            foreach (var handler in typeHandlers.ToList())
            {
                await handler(@event);
            }
        }
    }
    
    public IDisposable Subscribe<T>(Func<T, Task> handler) where T : IAgentEvent
    {
        var handlers = _handlers.GetOrAdd(typeof(T), _ => new List<Delegate>());
        handlers.Add(handler);
        
        return new Subscription(() => handlers.Remove(handler));
    }
}
```

### 1.5 Redis 事件总线（分布式）

```csharp
public class RedisEventBus : IEventBus
{
    private readonly IConnectionMultiplexer _redis;
    private readonly ISubscriber _subscriber;
    private readonly string _channel = "sre-agent-events";
    
    public async Task PublishAsync<T>(T @event) where T : IAgentEvent
    {
        var message = JsonSerializer.Serialize(@event, new JsonSerializerOptions
        {
            PropertyNamingPolicy = JsonNamingPolicy.CamelCase
        });
        
        await _subscriber.PublishAsync(_channel, message);
    }
    
    public IDisposable Subscribe<T>(Func<T, Task> handler) where T : IAgentEvent
    {
        var channel = _subscriber.Subscribe(_channel);
        
        channel.OnMessage(async msg =>
        {
            var @event = JsonSerializer.Deserialize<T>(msg.Message);
            if (@event != null)
            {
                await handler(@event);
            }
        });
        
        return new Subscription(() => channel.Unsubscribe());
    }
}
```

```csharp
public interface ISessionHub
{
    Task OnSessionStatusChanged(Guid sessionId, string status);
    Task OnAgentStarted(Guid sessionId, string agentId, string task);
    Task OnAgentCompleted(Guid sessionId, string agentId, AgentFinding finding);
    Task OnToolExecuting(Guid sessionId, string toolName, object parameters);
    Task OnToolCompleted(Guid sessionId, string toolName, ToolResult result);
    Task OnApprovalRequired(Guid sessionId, ApprovalRequest request);
    Task OnDiagnosisUpdated(Guid sessionId, DiagnosisUpdate update);
    Task OnMessageAdded(Guid sessionId, Message message);

public class SessionHub : Hub<ISessionHub>
{
    public async Task JoinSession(Guid sessionId)
    {
        await Groups.AddToGroupAsync(Context.ConnectionId, sessionId.ToString());
    }
    
    public async Task LeaveSession(Guid sessionId)
    {
        await Groups.RemoveFromGroupAsync(Context.ConnectionId, sessionId.ToString());
    }
}
```

### 3.1 指标收集

```csharp
public class AgentMetrics
{
    private readonly Meter _meter;
    
    // 计数器
    private readonly Counter<long> _sessionsCreated;
    private readonly Counter<long> _sessionsCompleted;
    private readonly Counter<long> _sessionsFailed;
    private readonly Counter<long> _toolInvocations;
    private readonly Counter<long> _approvalRequests;
    
    // 直方图
    private readonly Histogram<double> _sessionDuration;
    private readonly Histogram<double> _agentDuration;
    private readonly Histogram<double> _toolDuration;
    private readonly Histogram<int> _contextTokens;
    
    // 仪表
    private readonly ObservableGauge<int> _activeSessions;
    
    public AgentMetrics(IMeterFactory meterFactory)
    {
        _meter = meterFactory.Create("SreAgent");
        
        _sessionsCreated = _meter.CreateCounter<long>(
            "sre_agent_sessions_created_total",
            description: "Total number of sessions created");
            
        _sessionDuration = _meter.CreateHistogram<double>(
            "sre_agent_session_duration_seconds",
            description: "Session duration in seconds");
            
        _contextTokens = _meter.CreateHistogram<int>(
            "sre_agent_context_tokens",
            description: "Context token count");
            
        _activeSessions = _meter.CreateObservableGauge(
            "sre_agent_active_sessions",
            () => GetActiveSessionCount(),
            description: "Number of active sessions");
    }
    
    public void RecordSessionCreated(string alertType)
    {
        _sessionsCreated.Add(1, new("alert_type", alertType));
    }
    
    public void RecordSessionCompleted(string alertType, double durationSeconds, bool success)
    {
        if (success)
            _sessionsCompleted.Add(1, new("alert_type", alertType));
        else
            _sessionsFailed.Add(1, new("alert_type", alertType));
            
        _sessionDuration.Record(durationSeconds, 
            new("alert_type", alertType), 
            new("success", success));
    }
    
    public void RecordToolInvocation(string toolName, double durationMs, bool success)
    {
        _toolInvocations.Add(1, 
            new("tool", toolName), 
            new("success", success));
        _toolDuration.Record(durationMs / 1000.0, new("tool", toolName));
    }
}
```

### 3.4 健康检查

```csharp
public class AgentHealthCheck : IHealthCheck
{
    private readonly ISessionRepository _sessionRepository;
    private readonly IToolRegistry _toolRegistry;
    private readonly IChatClient _chatClient;
    
    public async Task<HealthCheckResult> CheckHealthAsync(
        HealthCheckContext context,
        CancellationToken cancellationToken = default)
    {
        var checks = new Dictionary<string, object>();
        var isHealthy = true;
        
        // 检查数据库连接
        try
        {
            await _sessionRepository.GetByStatusAsync(SessionStatus.Running);
            checks["database"] = "healthy";
        }
        catch (Exception ex)
        {
            checks["database"] = $"unhealthy: {ex.Message}";
            isHealthy = false;
        }
        
        // 检查 LLM 连接
        try
        {
            await _chatClient.CompleteAsync("test", cancellationToken: cancellationToken);
            checks["llm"] = "healthy";
        }
        catch (Exception ex)
        {
            checks["llm"] = $"unhealthy: {ex.Message}";
            isHealthy = false;
        }
        
        // 检查工具可用性
        var tools = _toolRegistry.GetAllTools();
        checks["tools_registered"] = tools.Count();
        
        return isHealthy
            ? HealthCheckResult.Healthy("All systems operational", checks)
            : HealthCheckResult.Unhealthy("Some systems are unhealthy", data: checks);
    }
}
```

## 4. 仪表板数据

### 4.1 统计查询

```csharp
public interface IAnalyticsService
{
    Task<SessionStats> GetSessionStatsAsync(DateTime from, DateTime to);
    Task<IEnumerable<AlertTypeStats>> GetAlertTypeStatsAsync(DateTime from, DateTime to);
    Task<IEnumerable<ToolUsageStats>> GetToolUsageStatsAsync(DateTime from, DateTime to);
}

public class SessionStats
{
    public int TotalSessions { get; set; }
    public int CompletedSessions { get; set; }
    public int FailedSessions { get; set; }
    public double AverageDurationSeconds { get; set; }
    public double AverageConfidence { get; set; }
    public int TotalToolInvocations { get; set; }
    public int ApprovalRequests { get; set; }
}

public class AnalyticsService : IAnalyticsService
{
    private readonly AppDbContext _context;
    
    public async Task<SessionStats> GetSessionStatsAsync(DateTime from, DateTime to)
    {
        var sessions = await _context.Sessions
            .Where(s => s.CreatedAt >= from && s.CreatedAt <= to)
            .ToListAsync();
            
        return new SessionStats
        {
            TotalSessions = sessions.Count,
            CompletedSessions = sessions.Count(s => s.Status == SessionStatus.Completed),
            FailedSessions = sessions.Count(s => s.Status == SessionStatus.Failed),
            AverageDurationSeconds = sessions
                .Where(s => s.CompletedAt.HasValue)
                .Average(s => (s.CompletedAt.Value - s.StartedAt.Value).TotalSeconds),
            AverageConfidence = sessions
                .Where(s => s.Confidence > 0)
                .Average(s => s.Confidence)
        };
    }
}
```
