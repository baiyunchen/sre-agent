# 执行过程持久化

## 1. 概述

持久化是 AI Agent 系统可靠性的关键保障，需要实现：
- 完整记录所有执行细节
- 支持断点恢复
- 支持人工中断和干预
- 提供审计追溯能力

## 2. 会话状态管理

### 2.1 会话状态机

```
┌─────────┐
│ Created │
└────┬────┘
     │ Start
     ▼
┌─────────┐     Approval     ┌──────────────────┐
│ Running │ ───────────────► │ WaitingApproval  │
└────┬────┘                  └────────┬─────────┘
     │                                │ Approved/Rejected
     │ ◄──────────────────────────────┘
     │
     ├─── Complete ───► ┌───────────┐
     │                  │ Completed │
     │                  └───────────┘
     │
     ├─── Error ──────► ┌────────┐
     │                  │ Failed │
     │                  └────────┘
     │
     ├─── Timeout ────► ┌──────────┐
     │                  │ TimedOut │
     │                  └──────────┘
     │
     └─── Interrupt ──► ┌─────────────┐
                        │ Interrupted │ (可恢复)
                        └─────────────┘
```

### 2.3 会话存储

```csharp
public interface ISessionRepository
{
    Task<Session> GetAsync(Guid id);
    Task<Session> CreateAsync(Session session);
    Task UpdateAsync(Session session);
    Task<IEnumerable<Session>> GetByStatusAsync(SessionStatus status);
    Task<IEnumerable<Session>> GetByAlertAsync(string alertId);
}

public class SessionRepository : ISessionRepository
{
    private readonly AppDbContext _context;
    
    public async Task<Session> GetAsync(Guid id)
    {
        return await _context.Sessions
            .Include(s => s.Messages)
            .Include(s => s.AgentRuns)
                .ThenInclude(r => r.ToolInvocations)
            .FirstOrDefaultAsync(s => s.Id == id);
    }
    
    public async Task UpdateAsync(Session session)
    {
        session.UpdatedAt = DateTime.UtcNow;
        _context.Sessions.Update(session);
        await _context.SaveChangesAsync();
    }
}
```

### 5.2 恢复服务

```csharp
public interface ISessionRecoveryService
{
    Task<Session> ResumeSessionAsync(Guid sessionId);
    Task<bool> CanResumeAsync(Guid sessionId);
}

public class SessionRecoveryService : ISessionRecoveryService
{
    private readonly ISessionRepository _sessionRepository;
    private readonly ICheckpointService _checkpointService;
    private readonly IAgentOrchestrator _orchestrator;
    
    public async Task<bool> CanResumeAsync(Guid sessionId)
    {
        var session = await _sessionRepository.GetAsync(sessionId);
        return session?.Status == SessionStatus.Interrupted 
            || session?.Status == SessionStatus.WaitingApproval;
    }
    
    public async Task<Session> ResumeSessionAsync(Guid sessionId)
    {
        var session = await _sessionRepository.GetAsync(sessionId);
        
        if (!await CanResumeAsync(sessionId))
        {
            throw new InvalidOperationException(
                $"Session {sessionId} cannot be resumed from status {session.Status}");
        }
        
        // 获取最新检查点
        var checkpoint = await _checkpointService.GetLatestCheckpointAsync(sessionId);
        
        if (checkpoint != null)
        {
            // 从检查点恢复
            session = await _checkpointService.RestoreFromCheckpointAsync(checkpoint.Id);
        }
        
        // 更新状态
        session.Status = SessionStatus.Running;
        await _sessionRepository.UpdateAsync(session);
        
        // 继续执行
        await _orchestrator.ResumeAsync(session);
        
        return session;
    }
}
```

### 6.2 干预服务

```csharp
public interface IInterventionService
{
    Task InterruptSessionAsync(Guid sessionId, string reason, string userId);
    Task CancelSessionAsync(Guid sessionId, string reason, string userId);
    Task ProvideInputAsync(Guid sessionId, JsonDocument input, string userId);
    Task OverrideDecisionAsync(Guid sessionId, JsonDocument decision, string userId);
}

public class InterventionService : IInterventionService
{
    public async Task InterruptSessionAsync(Guid sessionId, string reason, string userId)
    {
        var session = await _sessionRepository.GetAsync(sessionId);
        
        if (session.Status != SessionStatus.Running)
        {
            throw new InvalidOperationException("Can only interrupt running sessions");
        }
        
        // 创建检查点
        await _checkpointService.CreateCheckpointAsync(session, "interrupt");
        
        // 更新状态
        session.Status = SessionStatus.Interrupted;
        await _sessionRepository.UpdateAsync(session);
        
        // 记录干预
        await _interventionRepository.AddAsync(new Intervention
        {
            SessionId = sessionId,
            Type = InterventionType.Interrupt,
            Reason = reason,
            IntervenedBy = userId,
            IntervenedAt = DateTime.UtcNow
        });
        
        // 发布事件
        await _eventBus.PublishAsync(new SessionInterruptedEvent(sessionId, reason, userId));
    }
}
```

```csharp
public class AppDbContext : DbContext
{
    public DbSet<Session> Sessions { get; set; }
    public DbSet<Message> Messages { get; set; }
    public DbSet<AgentRun> AgentRuns { get; set; }
    public DbSet<ToolInvocation> ToolInvocations { get; set; }
    public DbSet<Checkpoint> Checkpoints { get; set; }
    public DbSet<Intervention> Interventions { get; set; }
    public DbSet<ApprovalRule> ApprovalRules { get; set; }
    
    protected override void OnModelCreating(ModelBuilder modelBuilder)
    {
        modelBuilder.Entity<Session>(entity =>
        {
            entity.HasKey(e => e.Id);
            entity.Property(e => e.AlertData).HasColumnType("jsonb");
            entity.Property(e => e.ExecutionState).HasColumnType("jsonb");
            entity.Property(e => e.Diagnosis).HasColumnType("jsonb");
            entity.HasIndex(e => e.Status);
            entity.HasIndex(e => e.AlertId);
            entity.HasIndex(e => e.CreatedAt);
        });
        
        modelBuilder.Entity<Message>(entity =>
        {
            entity.HasKey(e => e.Id);
            entity.Property(e => e.Metadata).HasColumnType("jsonb");
            entity.HasOne(e => e.Session)
                .WithMany(s => s.Messages)
                .HasForeignKey(e => e.SessionId);
            entity.HasIndex(e => new { e.SessionId, e.CreatedAt });
        });
        
        // ... 其他实体配置
    }
}
```

### 7.2 迁移脚本

```sql
-- 会话表
CREATE TABLE sessions (
    id UUID PRIMARY KEY,
    status VARCHAR(50) NOT NULL,
    alert_id VARCHAR(255),
    alert_name VARCHAR(255),
    alert_data JSONB,
    service_name VARCHAR(255),
    service_metadata JSONB,
    current_agent_id VARCHAR(100),
    current_step INTEGER DEFAULT 0,
    execution_state JSONB,
    diagnosis JSONB,
    diagnosis_summary TEXT,
    confidence DECIMAL(3,2),
    created_at TIMESTAMP NOT NULL,
    started_at TIMESTAMP,
    completed_at TIMESTAMP,
    updated_at TIMESTAMP NOT NULL
);

CREATE INDEX idx_sessions_status ON sessions(status);
CREATE INDEX idx_sessions_alert_id ON sessions(alert_id);
CREATE INDEX idx_sessions_created_at ON sessions(created_at);

-- 消息表
CREATE TABLE messages (
    id UUID PRIMARY KEY,
    session_id UUID NOT NULL REFERENCES sessions(id),
    role VARCHAR(50) NOT NULL,
    content TEXT,
    tool_calls JSONB,
    tool_call_id VARCHAR(255),
    metadata JSONB,
    estimated_tokens INTEGER,
    agent_id VARCHAR(100),
    source VARCHAR(100),
    created_at TIMESTAMP NOT NULL
);

CREATE INDEX idx_messages_session ON messages(session_id, created_at);

-- Agent 运行记录表
CREATE TABLE agent_runs (
    id UUID PRIMARY KEY,
    session_id UUID NOT NULL REFERENCES sessions(id),
    agent_id VARCHAR(100) NOT NULL,
    agent_name VARCHAR(255),
    input JSONB,
    output JSONB,
    status VARCHAR(50) NOT NULL,
    error_message TEXT,
    confidence DECIMAL(3,2),
    finding JSONB,
    started_at TIMESTAMP NOT NULL,
    completed_at TIMESTAMP,
    duration_ms BIGINT
);

CREATE INDEX idx_agent_runs_session ON agent_runs(session_id);

-- 工具调用记录表
CREATE TABLE tool_invocations (
    id UUID PRIMARY KEY,
    agent_run_id UUID NOT NULL REFERENCES agent_runs(id),
    tool_name VARCHAR(100) NOT NULL,
    parameters JSONB,
    result JSONB,
    approval_status VARCHAR(50),
    approved_by VARCHAR(100),
    approved_at TIMESTAMP,
    status VARCHAR(50) NOT NULL,
    error_message TEXT,
    requested_at TIMESTAMP NOT NULL,
    executed_at TIMESTAMP,
    completed_at TIMESTAMP,
    duration_ms BIGINT
);

CREATE INDEX idx_tool_invocations_agent_run ON tool_invocations(agent_run_id);

-- 检查点表
CREATE TABLE checkpoints (
    id UUID PRIMARY KEY,
    session_id UUID NOT NULL REFERENCES sessions(id),
    checkpoint_name VARCHAR(255),
    step_number INTEGER,
    session_state JSONB,
    context_state JSONB,
    agent_state JSONB,
    created_at TIMESTAMP NOT NULL
);

CREATE INDEX idx_checkpoints_session ON checkpoints(session_id, created_at);
```

```csharp
public interface IAuditService
{
    Task LogAsync(AuditLog log);
    Task<IEnumerable<AuditLog>> GetBySessionAsync(Guid sessionId);
}

// 通过事件订阅自动记录审计日志
public class AuditEventHandler : 
    IEventHandler<ToolExecutionCompletedEvent>,
    IEventHandler<ApprovalDecisionMadeEvent>,
    IEventHandler<SessionInterruptedEvent>
{
    private readonly IAuditService _auditService;
    
    public async Task HandleAsync(ToolExecutionCompletedEvent @event)
    {
        await _auditService.LogAsync(new AuditLog
        {
            SessionId = @event.Invocation.SessionId,
            EventType = "ToolExecuted",
            EventDescription = $"Tool {@event.Invocation.ToolName} executed",
            EventData = JsonDocument.Parse(JsonSerializer.Serialize(@event)),
            Actor = "agent",
            ActorId = @event.Invocation.AgentId,
            OccurredAt = DateTime.UtcNow
        });
    }
}
```
