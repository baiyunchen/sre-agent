# 上下文管理策略

## 1. 概述

上下文管理是 AI Agent 系统的核心挑战之一。Agent Framework 提供：
- **抽象接口**：`IContextManager` 定义标准契约
- **默认实现**：`DefaultContextManager` 满足通用需求
- **可扩展性**：业务层可自定义实现

## 2. 数据模型：Session - Message - Part

### 2.1 三层结构

```
Session (会话)
    │
    ├── Message (消息)
    │       │
    │       ├── Part (文本部分)
    │       ├── Part (工具调用)
    │       └── Part (工具结果)
    │
    ├── Message
    │       │
    │       └── Part (文本部分)
    │
    └── ...
```

### 2.2 Message 定义

```csharp
/// <summary>
/// 消息 - 对话中的一条消息
/// </summary>
public class Message
{
    public Guid Id { get; set; } = Guid.NewGuid();
    public MessageRole Role { get; set; }
    public List<PartBase> Parts { get; set; } = new();
    public MessageMetadata Metadata { get; set; } = new();
    public DateTime CreatedAt { get; set; } = DateTime.UtcNow;
}

public enum MessageRole
{
    System,
    User,
    Assistant,
    Tool
}

public class MessageMetadata
{
    /// <summary>来源 Agent ID</summary>
    public string? AgentId { get; set; }
    
    /// <summary>预估 Token 数</summary>
    public int EstimatedTokens { get; set; }
    
    /// <summary>优先级（裁剪时参考）</summary>
    public int Priority { get; set; } = MessagePriority.Normal;
    
    /// <summary>是否可删除</summary>
    public bool IsDeletable { get; set; } = true;
    
    /// <summary>标签（用于分类和过滤）</summary>
    public HashSet<string> Tags { get; set; } = new();
    
    /// <summary>自定义扩展数据</summary>
    public Dictionary<string, object> Extra { get; set; } = new();
}

public static class MessagePriority
{
    public const int Critical = 100;    // 绝不删除
    public const int Important = 80;    // 高优先级保留
    public const int Normal = 50;       // 正常优先级
    public const int Low = 20;          // 低优先级
    public const int Disposable = 0;    // 可丢弃
}
```

## 3. 上下文管理器接口

### 3.1 接口定义

```csharp
/// <summary>
/// 上下文管理器接口
/// Framework 提供默认实现，业务层可扩展
/// </summary>
public interface IContextManager
{
    /// <summary>添加消息到上下文</summary>
    Task AddMessageAsync(Message message, CancellationToken cancellationToken = default);
    
    /// <summary>批量添加消息</summary>
    Task AddMessagesAsync(IEnumerable<Message> messages, CancellationToken cancellationToken = default);
    
    /// <summary>
    /// 获取用于发送给 LLM 的消息列表
    /// 可能经过裁剪以满足 Token 限制
    /// </summary>
    Task<IReadOnlyList<Message>> GetMessagesForCompletionAsync(
        int maxTokens,
        CancellationToken cancellationToken = default);
    
    /// <summary>获取完整消息历史（未裁剪）</summary>
    Task<IReadOnlyList<Message>> GetAllMessagesAsync(
        CancellationToken cancellationToken = default);
    
    /// <summary>当前 Token 使用量估算</summary>
    int EstimatedTokenCount { get; }
    
    /// <summary>清空上下文</summary>
    Task ClearAsync(CancellationToken cancellationToken = default);
    
    /// <summary>设置 System 消息（会替换已有的）</summary>
    Task SetSystemMessageAsync(string content, CancellationToken cancellationToken = default);
}
```

### 3.2 配置选项

```csharp
/// <summary>
/// 上下文管理器配置选项
/// </summary>
public class ContextManagerOptions
{
    /// <summary>默认最大 Token 数</summary>
    public int DefaultMaxTokens { get; set; } = 100000;
    
    /// <summary>触发裁剪的阈值（占 MaxTokens 的比例）</summary>
    public double TrimThreshold { get; set; } = 0.8;
    
    /// <summary>裁剪后的目标（占 MaxTokens 的比例）</summary>
    public double TrimTarget { get; set; } = 0.6;
    
    /// <summary>裁剪策略</summary>
    public TrimStrategy TrimStrategy { get; set; } = TrimStrategy.RemoveOldest;
    
    /// <summary>是否自动压缩长工具结果</summary>
    public bool AutoCompressToolResults { get; set; } = true;
    
    /// <summary>工具结果压缩阈值（Token 数）</summary>
    public int ToolResultCompressThreshold { get; set; } = 1000;
}

public enum TrimStrategy
{
    /// <summary>移除最早的消息</summary>
    RemoveOldest,
    
    /// <summary>按优先级移除</summary>
    RemoveByPriority,
    
    /// <summary>总结历史消息</summary>
    Summarize,
    
    /// <summary>滑动窗口</summary>
    SlidingWindow
}
```

## 4. 默认实现

### 4.1 DefaultContextManager

```csharp
/// <summary>
/// 默认上下文管理器实现
/// </summary>
public class DefaultContextManager : IContextManager
{
    private readonly List<Message> _messages = new();
    private readonly ITokenEstimator _tokenEstimator;
    private readonly ContextManagerOptions _options;
    private Message? _systemMessage;
    private readonly object _lock = new();
    
    public DefaultContextManager(
        ITokenEstimator tokenEstimator,
        ContextManagerOptions? options = null)
    {
        _tokenEstimator = tokenEstimator;
        _options = options ?? new ContextManagerOptions();
    }
    
    public int EstimatedTokenCount
    {
        get
        {
            lock (_lock)
            {
                var total = _systemMessage != null 
                    ? _systemMessage.Metadata.EstimatedTokens 
                    : 0;
                return total + _messages.Sum(m => m.Metadata.EstimatedTokens);
            }
        }
    }
    
    public Task SetSystemMessageAsync(string content, CancellationToken cancellationToken = default)
    {
        lock (_lock)
        {
            _systemMessage = new Message
            {
                Role = MessageRole.System,
                Parts = new List<PartBase> { new TextPart { Text = content } },
                Metadata = new MessageMetadata
                {
                    IsDeletable = false,
                    Priority = MessagePriority.Critical,
                    EstimatedTokens = _tokenEstimator.EstimateTokens(content)
                }
            };
        }
        return Task.CompletedTask;
    }
    
    public Task AddMessageAsync(Message message, CancellationToken cancellationToken = default)
    {
        // 计算 Token 数
        message.Metadata.EstimatedTokens = _tokenEstimator.EstimateTokens(message);
        
        // 自动压缩长工具结果
        if (_options.AutoCompressToolResults)
        {
            CompressToolResultsIfNeeded(message);
        }
        
        lock (_lock)
        {
            _messages.Add(message);
        }
        
        return Task.CompletedTask;
    }
    
    public Task AddMessagesAsync(IEnumerable<Message> messages, CancellationToken cancellationToken = default)
    {
        foreach (var message in messages)
        {
            AddMessageAsync(message, cancellationToken);
        }
        return Task.CompletedTask;
    }
    
    public Task<IReadOnlyList<Message>> GetMessagesForCompletionAsync(
        int maxTokens,
        CancellationToken cancellationToken = default)
    {
        lock (_lock)
        {
            var allMessages = BuildMessageList();
            var currentTokens = allMessages.Sum(m => m.Metadata.EstimatedTokens);
            
            // 如果未超限，直接返回
            if (currentTokens <= maxTokens)
            {
                return Task.FromResult<IReadOnlyList<Message>>(allMessages);
            }
            
            // 需要裁剪
            var trimmed = TrimMessages(allMessages, maxTokens);
            return Task.FromResult<IReadOnlyList<Message>>(trimmed);
        }
    }
    
    public Task<IReadOnlyList<Message>> GetAllMessagesAsync(
        CancellationToken cancellationToken = default)
    {
        lock (_lock)
        {
            return Task.FromResult<IReadOnlyList<Message>>(BuildMessageList());
        }
    }
    
    public Task ClearAsync(CancellationToken cancellationToken = default)
    {
        lock (_lock)
        {
            _messages.Clear();
            // 注意：不清除 System 消息
        }
        return Task.CompletedTask;
    }
    
    private List<Message> BuildMessageList()
    {
        var result = new List<Message>();
        if (_systemMessage != null)
        {
            result.Add(_systemMessage);
        }
        result.AddRange(_messages);
        return result;
    }
    
    private List<Message> TrimMessages(List<Message> messages, int maxTokens)
    {
        return _options.TrimStrategy switch
        {
            TrimStrategy.RemoveOldest => TrimByRemovingOldest(messages, maxTokens),
            TrimStrategy.RemoveByPriority => TrimByPriority(messages, maxTokens),
            TrimStrategy.SlidingWindow => TrimBySlidingWindow(messages, maxTokens),
            _ => TrimByRemovingOldest(messages, maxTokens)
        };
    }
    
    private List<Message> TrimByRemovingOldest(List<Message> messages, int maxTokens)
    {
        var result = new List<Message>();
        var currentTokens = 0;
        
        // 始终保留 System 消息
        var systemMsg = messages.FirstOrDefault(m => m.Role == MessageRole.System);
        if (systemMsg != null)
        {
            result.Add(systemMsg);
            currentTokens += systemMsg.Metadata.EstimatedTokens;
        }
        
        // 从后向前遍历，保留最新的消息
        var nonSystemMessages = messages
            .Where(m => m.Role != MessageRole.System)
            .Reverse()
            .ToList();
        
        foreach (var msg in nonSystemMessages)
        {
            if (currentTokens + msg.Metadata.EstimatedTokens > maxTokens)
            {
                // 如果是不可删除的消息，强制保留
                if (!msg.Metadata.IsDeletable)
                {
                    result.Insert(result.Count > 0 && result[0].Role == MessageRole.System ? 1 : 0, msg);
                    currentTokens += msg.Metadata.EstimatedTokens;
                }
                continue;
            }
            
            result.Insert(result.Count > 0 && result[0].Role == MessageRole.System ? 1 : 0, msg);
            currentTokens += msg.Metadata.EstimatedTokens;
        }
        
        return result;
    }
    
    private List<Message> TrimByPriority(List<Message> messages, int maxTokens)
    {
        // 按优先级排序，优先保留高优先级消息
        var sorted = messages
            .OrderByDescending(m => m.Metadata.Priority)
            .ThenByDescending(m => m.CreatedAt)
            .ToList();
        
        var result = new List<Message>();
        var currentTokens = 0;
        
        foreach (var msg in sorted)
        {
            if (currentTokens + msg.Metadata.EstimatedTokens <= maxTokens)
            {
                result.Add(msg);
                currentTokens += msg.Metadata.EstimatedTokens;
            }
        }
        
        // 按时间重新排序
        return result.OrderBy(m => m.CreatedAt).ToList();
    }
    
    private List<Message> TrimBySlidingWindow(List<Message> messages, int maxTokens)
    {
        var result = new List<Message>();
        
        // 保留 System 消息
        var systemMsg = messages.FirstOrDefault(m => m.Role == MessageRole.System);
        if (systemMsg != null)
        {
            result.Add(systemMsg);
        }
        
        // 保留最近 N 条消息
        var nonSystemMessages = messages
            .Where(m => m.Role != MessageRole.System)
            .ToList();
        
        var remainingTokens = maxTokens - (systemMsg?.Metadata.EstimatedTokens ?? 0);
        var windowMessages = new List<Message>();
        
        for (int i = nonSystemMessages.Count - 1; i >= 0; i--)
        {
            var msg = nonSystemMessages[i];
            if (windowMessages.Sum(m => m.Metadata.EstimatedTokens) + msg.Metadata.EstimatedTokens <= remainingTokens)
            {
                windowMessages.Insert(0, msg);
            }
            else
            {
                break;
            }
        }
        
        result.AddRange(windowMessages);
        return result;
    }
    
    private void CompressToolResultsIfNeeded(Message message)
    {
        foreach (var part in message.Parts.OfType<ToolResultPart>().ToList())
        {
            var tokens = _tokenEstimator.EstimateTokens(part.Content);
            if (tokens > _options.ToolResultCompressThreshold)
            {
                // 压缩长结果
                var compressed = CompressContent(part.Content, _options.ToolResultCompressThreshold);
                var index = message.Parts.IndexOf(part);
                message.Parts[index] = part with { Content = compressed };
            }
        }
    }
    
    private string CompressContent(string content, int targetTokens)
    {
        // 简单的截断策略，实际可以更智能
        var lines = content.Split('\n');
        var result = new List<string>();
        var currentTokens = 0;
        
        foreach (var line in lines)
        {
            var lineTokens = _tokenEstimator.EstimateTokens(line);
            if (currentTokens + lineTokens > targetTokens)
            {
                result.Add("... [content truncated] ...");
                break;
            }
            result.Add(line);
            currentTokens += lineTokens;
        }
        
        return string.Join('\n', result);
    }
}
```

## 5. Token 估算器

### 5.1 接口定义

```csharp
/// <summary>
/// Token 估算器接口
/// </summary>
public interface ITokenEstimator
{
    int EstimateTokens(string text);
    int EstimateTokens(Message message);
    int EstimateTokens(IEnumerable<Message> messages);
}
```

### 5.2 实现

```csharp
/// <summary>
/// 基于 Tiktoken 的 Token 估算器
/// </summary>
public class TiktokenEstimator : ITokenEstimator
{
    private readonly TikToken _encoder;
    
    public TiktokenEstimator(string model = "gpt-4")
    {
        _encoder = TikToken.EncodingForModel(model);
    }
    
    public int EstimateTokens(string text)
    {
        if (string.IsNullOrEmpty(text)) return 0;
        return _encoder.Encode(text).Count;
    }
    
    public int EstimateTokens(Message message)
    {
        // 消息格式化开销约 4 tokens
        const int messageOverhead = 4;
        
        var contentTokens = 0;
        foreach (var part in message.Parts)
        {
            contentTokens += part switch
            {
                TextPart text => EstimateTokens(text.Text),
                ToolCallPart toolCall => EstimateTokens(toolCall.Name) + EstimateTokens(toolCall.Arguments),
                ToolResultPart toolResult => EstimateTokens(toolResult.Content),
                _ => 0
            };
        }
        
        return messageOverhead + contentTokens;
    }
    
    public int EstimateTokens(IEnumerable<Message> messages)
    {
        return messages.Sum(EstimateTokens);
    }
}

/// <summary>
/// 简单的字符估算器（用于测试或无法使用 Tiktoken 时）
/// </summary>
public class SimpleTokenEstimator : ITokenEstimator
{
    private const double CharsPerToken = 4.0;
    
    public int EstimateTokens(string text)
    {
        if (string.IsNullOrEmpty(text)) return 0;
        return (int)Math.Ceiling(text.Length / CharsPerToken);
    }
    
    public int EstimateTokens(Message message)
    {
        return message.Parts.Sum(part => part switch
        {
            TextPart text => EstimateTokens(text.Text),
            ToolCallPart toolCall => EstimateTokens(toolCall.Name) + EstimateTokens(toolCall.Arguments),
            ToolResultPart toolResult => EstimateTokens(toolResult.Content),
            _ => 0
        }) + 4;
    }
    
    public int EstimateTokens(IEnumerable<Message> messages)
    {
        return messages.Sum(EstimateTokens);
    }
}
```

## 6. 业务层扩展示例

### 6.1 带总结的上下文管理器

```csharp
/// <summary>
/// 支持历史消息总结的上下文管理器
/// </summary>
public class SummarizingContextManager : IContextManager
{
    private readonly IContextManager _inner;
    private readonly IChatClient _chatClient;
    private readonly ITokenEstimator _tokenEstimator;
    private readonly SummarizingOptions _options;
    
    private Message? _summaryMessage;
    
    public async Task<IReadOnlyList<Message>> GetMessagesForCompletionAsync(
        int maxTokens,
        CancellationToken cancellationToken = default)
    {
        var allMessages = await _inner.GetAllMessagesAsync(cancellationToken);
        var currentTokens = allMessages.Sum(m => m.Metadata.EstimatedTokens);
        
        if (currentTokens <= maxTokens)
        {
            return allMessages;
        }
        
        // 需要总结历史消息
        var (messagesToSummarize, recentMessages) = SplitMessages(allMessages, maxTokens);
        
        if (messagesToSummarize.Any())
        {
            _summaryMessage = await GenerateSummaryAsync(messagesToSummarize, cancellationToken);
        }
        
        var result = new List<Message>();
        
        // System 消息
        var systemMsg = allMessages.FirstOrDefault(m => m.Role == MessageRole.System);
        if (systemMsg != null)
        {
            result.Add(systemMsg);
        }
        
        // 总结消息
        if (_summaryMessage != null)
        {
            result.Add(_summaryMessage);
        }
        
        // 近期消息
        result.AddRange(recentMessages);
        
        return result;
    }
    
    private async Task<Message> GenerateSummaryAsync(
        IEnumerable<Message> messages,
        CancellationToken cancellationToken)
    {
        var conversationText = FormatMessagesForSummary(messages);
        
        var prompt = $"""
            请总结以下对话的关键信息，保留：
            1. 重要的发现和结论
            2. 做出的关键决策
            3. 待处理的问题
            4. 相关的工具调用结果摘要
            
            对话内容：
            {conversationText}
            
            请用简洁的中文输出总结。
            """;
        
        var response = await _chatClient.CompleteAsync(prompt, cancellationToken: cancellationToken);
        
        return new Message
        {
            Role = MessageRole.System,
            Parts = new List<PartBase>
            {
                new TextPart { Text = $"[历史对话总结]\n{response.Message.Text}" }
            },
            Metadata = new MessageMetadata
            {
                Priority = MessagePriority.Important,
                IsDeletable = false,
                Tags = new HashSet<string> { "summary" },
                EstimatedTokens = _tokenEstimator.EstimateTokens(response.Message.Text ?? "")
            }
        };
    }
    
    // ... 其他方法委托给 _inner
}
```

```csharp
/// <summary>
/// 支持持久化的上下文管理器
/// </summary>
public class PersistentContextManager : IContextManager
{
    private readonly IContextManager _inner;
    private readonly IMessageRepository _repository;
    private readonly Guid _sessionId;
    
    public async Task AddMessageAsync(Message message, CancellationToken cancellationToken = default)
    {
        // 添加到内存
        await _inner.AddMessageAsync(message, cancellationToken);
        
        // 持久化
        await _repository.SaveAsync(_sessionId, message, cancellationToken);
    }
    
    public async Task<IReadOnlyList<Message>> GetAllMessagesAsync(
        CancellationToken cancellationToken = default)
    {
        // 优先从内存获取
        var inMemory = await _inner.GetAllMessagesAsync(cancellationToken);
        if (inMemory.Any())
        {
            return inMemory;
        }
        
        // 从数据库加载
        var fromDb = await _repository.GetBySessionAsync(_sessionId, cancellationToken);
        await _inner.AddMessagesAsync(fromDb, cancellationToken);
        
        return fromDb;
    }
    
    // ... 其他方法
}
```

### 7.2 Skill 加载工具

```csharp
/// <summary>
/// 加载 Skill 的工具
/// </summary>
public class LoadSkillTool : ITool
{
    private readonly ISkillRepository _skillRepository;
    
    public string Name => "load_skill";
    public string Description => "Load detailed instructions for a specific skill when needed";
    
    public JsonElement ParameterSchema => JsonDocument.Parse("""
        {
            "type": "object",
            "properties": {
                "skill_id": {
                    "type": "string",
                    "description": "The ID of the skill to load"
                }
            },
            "required": ["skill_id"]
        }
        """).RootElement;
    
    public async Task<ToolResult> ExecuteAsync(
        ToolExecutionContext context,
        CancellationToken cancellationToken = default)
    {
        try
        {
            var skillId = context.Parameters.GetProperty("skill_id").GetString();
            if (string.IsNullOrEmpty(skillId))
            {
                return ToolResult.Failure("skill_id is required", "INVALID_PARAM");
            }
            
            var skill = await _skillRepository.GetAsync(skillId, cancellationToken);
            if (skill == null)
            {
                return ToolResult.Failure($"Skill '{skillId}' not found", "NOT_FOUND");
            }
            
            return ToolResult.Success(skill.DetailedInstructions, skill);
        }
        catch (Exception ex)
        {
            return ToolResult.FromException(ex);
        }
    }
}
```

```yaml
context_manager:
```
